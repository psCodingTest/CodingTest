# 백준 2468번: 안전 영역
* https://www.acmicpc.net/problem/2468
<br>

## 시간복잡도
* 2 <= N <= 100
* 1 <= H <= 100
* O(H × N²) = O(1,000,000)
<br>

## 풀이
* 브루트포스
* 완전탐색

<br>

## 어려웠던 부분
* 이중 배열 딥카피하자

<br> 

## 코드
```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Arrays;

public class Main {
    static int[] dy = {0, 1, 0, -1};
    static int[] dx = {1, 0, -1, 0};
    static int N = 0;

    public static void main(String[] args) throws IOException {
        //장마철에 물에 잠기지 않는 안전한 영역의 최대 개수
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        N = Integer.parseInt(br.readLine());
        int[][] map = new int[N][N];
        int maxHeight = 0;
        int answer = 1;

        for (int i = 0; i < N; i++) {
            String[] s = br.readLine().split(" ");
            int[] array = Arrays.stream(s).mapToInt(Integer::parseInt).toArray();
            map[i] = array;
            maxHeight = Arrays.stream(array).max().getAsInt();
        }

        for (int h = 1; h <= maxHeight; h++) {
            //int[][] 딥카피
            int[][] tempMap = new int[N][];
            for (int i = 0; i < N; i++) {
                tempMap[i] = map[i].clone();
            }

            //높이에 따라 물 잠김
            for (int y = 0; y < N; y++) {
                for (int x = 0; x < N; x++) {
                    if (tempMap[y][x] <= h) {
                        tempMap[y][x] = -1;  //물에 잠긴 칸
                    }
                }
            }

            //안전한 영역 개수 세기
            int count = 0;
            for (int y = 0; y < N; y++) {
                for (int x = 0; x < N; x++) {
                    if (tempMap[y][x] != -1) {
                        count++;
                        dfs(y, x, tempMap);
                    }
                }

            }
            answer = Math.max(answer, count);
        }

        System.out.println(answer);
    }

    public static void dfs(int y, int x, int[][] tempMap) {
        tempMap[y][x] = -1;

        for (int i = 0; i < 4; i++) {
            int ny = y + dy[i];
            int nx = x + dx[i];

            if (0 <= ny && ny < N && 0 <= nx && nx < N) {
                if (tempMap[ny][nx] != -1) {
                    dfs(ny, nx, tempMap);
                }
            }

        }
    }
}
```
