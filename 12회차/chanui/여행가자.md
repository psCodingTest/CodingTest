#include <iostream>
using namespace std;

int N,M;
int parent[201]{};

int find(int k) {
    if (k == parent[k]) return k;
    return parent[k] = find(parent[k]);
}

void merge(int a, int b) {
    a = find(a);
    b = find(b);
    parent[a] = b;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cin>>N>>M;
    int tmp;
    for(int i=0; i<=N; i++) parent[i] = i;
    for(int i=0; i<N; i++) {
        for(int j=0; j<N; j++) {
            cin>>tmp;
            if(tmp==1) {
                if(find(i) == find(j)) continue;
                merge(i, j);
            }
        }
    }

    bool chk = true;
    int chkCity;
    for(int i=0; i<M; i++) {
        cin>>tmp;
        if(i==0) {
            chkCity = tmp-1;
            continue;
        }
        if(find(chkCity) != find(tmp-1)) {
            chk = false;
            break;
        }
    }
    chk ? cout<<"YES" : cout<<"NO";
}
