# 주사위 쌓기

- https://www.acmicpc.net/problem/2116

<br>

## 시간복잡도


## 코드

```java

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

public class Main {
    static int[][] dices;
    static int N;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        N = Integer.parseInt(br.readLine());
        dices = new int[N][6];
        for (int i = 0; i < N; i++) {
            StringTokenizer st = new StringTokenizer(br.readLine());
            for (int j = 0; j < 6; j++) {
                int diceScale = Integer.parseInt(st.nextToken());
                dices[i][j] = diceScale;
            }
        }
        int result = 0;
        for (int i = 1; i <= 6; i++) {
            result = Math.max(dfs(i, 0, 0), result);
        }
        System.out.println(result);
    }

    private static int dfs(int bottomValue, int r, int value) {
        if (r == N) {
            return value;
        }
        int bottomIdx = -1;
        for (int i = 0; i < 6; i++) {
            if (dices[r][i] == bottomValue) {
                bottomIdx = i;
                break;
            }
        }

        int topIdx = getTopIdx(bottomIdx);
        int topValue = dices[r][topIdx];

        int MaxValue = findMaxValue(bottomValue, topValue);
        return dfs(topValue, r + 1, value + MaxValue);
    }

    private static int findMaxValue(int bottom, int top) {
        for (int i = 6; i >= 1; i--) {
            if (bottom == i || top == i) continue;
            return i;
        }
        return 0;
    }

    private static int getTopIdx(int idx) {
        switch (idx) {
            case 0:
                return 5;
            case 1:
                return 3;
            case 2:
                return 4;
            case 3:
                return 1;
            case 4:
                return 2;
            case 5:
                return 0;
            default:
                return -1;
        }
    }
}
```
