# 개똥벌레

- https://www.acmicpc.net/problem/3020

<br>

## 시간복잡도
O(N Log N)

## 코드
이분탐색

```java
import java.util.*;
import java.io.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        String[] str = br.readLine().split(" ");
        int n = Integer.parseInt(str[0]);
        int h = Integer.parseInt(str[1]);

        int[] bottom = new int[n/2];
        int[] top = new int[n/2];
        for (int i = 0; i < n/2; i++) {
            bottom[i] = Integer.parseInt(br.readLine()); // 짝수
            top[i] = Integer.parseInt(br.readLine()); // 홀수
        }

        Arrays.sort(bottom);
        Arrays.sort(top);

        int min = Integer.MAX_VALUE ;
        int count = 0;

        for (int i = 1; i <= h; i++) { // i: 구간
            int cnt1 = bottom.length - search(bottom, i);
            int cnt2 = top.length - search(top, h-i+1);

            int sum = cnt1 + cnt2;
            if (sum < min) {
                min = sum;
                count = 1;
            } else if (sum == min) {
                count++;
            }

        }
        System.out.println(min + " " + count);
    }

    //target 이상인 값의 개수
    static public int search(int[] arr, int target) {
        int left = 0, right = arr.length;

        while (left < right) {
            int mid = (left + right) / 2;
            if (arr[mid] < target) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }

        return left;
    }
}
```
