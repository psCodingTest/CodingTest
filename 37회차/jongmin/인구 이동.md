# 인구 이동

- https://www.acmicpc.net/problem/16234

<br>

## 시간복잡도
O(N^4)
## 코드

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayDeque;
import java.util.Arrays;
import java.util.Queue;
import java.util.StringTokenizer;

public class Main {
    static int[][] moves = new int[][] {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
    static boolean[][] visited;
    static int[][] arr;
    static int L, R, N;
    static boolean isMoved = false;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());

        N = Integer.parseInt(st.nextToken());
        L = Integer.parseInt(st.nextToken());
        R = Integer.parseInt(st.nextToken());
        arr = new int[N][N];

        for (int i = 0; i < N; i++) {
            arr[i] = Arrays.stream(br.readLine().split(" ")).mapToInt(Integer::parseInt).toArray();
        }
        int answer = -1;
        do {
            answer++;
            visited = new boolean[N][N];
            isMoved = false;

            for (int i = 0; i < N; i++) {
                for (int j = 0; j < N; j++) {
                    if (visited[i][j]) continue;
                    Queue<int[]> open = new ArrayDeque<>();
                    bfs(i, j, open);
                }
            }
        } while (isMoved);
        System.out.println(answer);
    }

    private static void bfs(int x, int y, Queue<int[]> open) {
        Queue<int[]> q = new ArrayDeque<>();
        if (visited[x][y]) return;
        visited[x][y] = true;
        open.offer(new int[] {x, y});
        q.offer(new int[] {x, y});
        int sum = arr[x][y];
        int cnt = 1;
        while (!q.isEmpty()) {
            int[] curr = q.poll();
            int cx = curr[0];
            int cy = curr[1];
            for (int[] move : moves) {
                int nx = move[0] + cx;
                int ny = move[1] + cy;
                if (nx < 0 || ny < 0 || nx >= N || ny >= N) continue;
                int diff = Math.abs(arr[cx][cy] - arr[nx][ny]);
                if (!visited[nx][ny] && diff >= L && diff <= R) {
                    q.offer(new int[] {nx, ny});
                    open.offer(new int[] {nx, ny});
                    visited[nx][ny] = true;
                    sum += arr[nx][ny];
                    cnt++;
                    isMoved = true;
                }
            }
        }

        if (open.size() > 1) {
            while (!open.isEmpty()) {
                int[] curr = open.poll();
                int cx = curr[0];
                int cy = curr[1];
                arr[cx][cy] = sum / cnt;
            }
        }
    }
}
```
